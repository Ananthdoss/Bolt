630  {
631  static_assert( std::is_same< array_type, container_type >::value,
632  "This member function is only valid for concurrency::array types.");

634 
635  if( reqSize == cap )
636  return;

638  //TODO - Add if statement for max size allowed in array class
639  container_type* l_tmpBuffer = new container_type((int)reqSize);
640  if( m_Size > 0 )
641  {
642  //1622 Arrays are logically considered to be value types in that when an array is copied to another array,
643  //a deep copy is performed. Two arrays never point to the same data.
644  //m_Size data elements are copied
645 
646  if( reqSize > m_Size )
647  {

649  arrayview_type l_tmpBufferSectionAV =
650  l_tmpBuffer->section((int)m_Size, (int)(reqSize - m_Size));

652  {
653  l_tmpBufferSectionAV[idx] = val;

656  else
657  {
658  arrayview_type l_devMemoryAV = m_devMemory->section(0, (int)reqSize);
659  arrayview_type l_tmpBufferAV = l_tmpBuffer->section(0, (int)reqSize);
660  l_devMemoryAV.copy_to(l_tmpBufferAV);
661  }
662  }

664  {
665  arrayview_type l_tmpBufferAV(*l_tmpBuffer);
666  Concurrency::parallel_for_each(l_tmpBufferAV.extent, [=](Concurrency::index<1> idx) restrict(amp)

668  l_tmpBufferAV[idx] = val;
669  });
670  }
671
